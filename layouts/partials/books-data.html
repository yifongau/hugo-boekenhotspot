{{/* This script makes the assumption that every ebook file has a corresponding book cover and metadata file. Build a slice of book objects from files in assets/boeken. Each object contains: - bookFilePath - imageFilePath - metadata (parsed OPF) */}} {{- $books := slice -}} {{/* File type matchers */}} {{- $bookExtRegex := `(?i)\.(epub|pdf|djvu|fb2|cbz)$` -}} {{- $imageExtRegex := `(?i)\.(jpg|jpeg|png|webp)$` -}} {{- $metadataExtRegex := `(?i)\.(opf)$` -}} {{/* Load all candidate files once (single pass input) */}} {{- $allFiles := resources.Match "boeken/**" -}} {{/* Initialize hash map to keep track of already seen books */}} {{- $byKey := dict -}} {{- range $allFiles -}} {{- $dir := path.Dir .Name -}} {{- $base := path.Base .Name -}} {{- $stem := "" -}} {{/* Check extensions */}} {{- if findRE $bookExtRegex .Name -}} {{- $stem = replaceRE $bookExtRegex "" $base -}} {{- else if findRE $imageExtRegex .Name -}} {{- $stem = replaceRE $imageExtRegex "" $base -}} {{- else if findRE $metadataExtRegex .Name -}} {{- $stem = replaceRE $metadataExtRegex "" $base -}} {{- end -}} {{- if ne $stem "" -}} {{/* Only iterate over file with valid file extension */}} {{- $key := printf "%s|%s" $dir $stem -}} {{/* If hashmap does not contain book yet, create empty record. */}} {{- if not (isset $byKey $key) -}} {{- $byKey = merge $byKey (dict $key (dict)) -}} {{- end -}} {{/* Fetch current book */}} {{- $entry := index $byKey $key -}} {{/* Book file: sanitize path, publish, store permalink and file extension */}} {{- if findRE $bookExtRegex .Name -}} {{- $safeAssetPath := replace .Name "%" "%25" -}} {{- $publishedBook := resources.Copy $safeAssetPath . -}} {{- $ext := path.Ext $publishedBook.RelPermalink -}} {{- $entry = merge $entry (dict "bookFilePath" $publishedBook.RelPermalink "ext" $ext) -}} {{- end -}} {{/* Book covers: sanitize path, publish, store permalink */}} {{- if findRE $imageExtRegex .Name -}} {{/* Convert to WebP and cap at 480w */}} {{- $processedImage := . -}} {{- if gt .Width 480 -}} {{- $processedImage = .Resize "480x webp q78" -}} {{- else -}} {{- $resizeSpec := printf "%dx webp q78" .Width -}} {{- $processedImage = .Resize $resizeSpec -}} {{- end -}} {{- $safeAssetPath := replace .Name "%" "%25" -}} {{- $safeWebpPath := replaceRE $imageExtRegex ".webp" $safeAssetPath -}} {{- $publishedImage := resources.Copy $safeWebpPath $processedImage -}} {{- $entry = merge $entry (dict "imageFilePath" $publishedImage.RelPermalink) -}} {{- end -}} {{/* Book metadata: parse OPF XML and store parsed object */}} {{- if findRE $metadataExtRegex .Name -}} {{- $parsedMetadata := transform.Unmarshal .Content -}} {{- $entry = merge $entry (dict "metadata" $parsedMetadata) -}} {{- end -}} {{/* Write resulting entry into lookup map */}} {{- $byKey = merge $byKey (dict $key $entry) -}} {{- end -}} {{- end -}} {{/* Range over resulting map. Only keep entries with .bookFilePath */}} {{- range $k, $entry := $byKey -}} {{- if and (isset $entry "bookFilePath") (isset $entry "imageFilePath") (isset $entry "metadata") -}} {{- $books = $books | append $entry -}} {{- else -}} {{- warnf "Incomplete book entry '%s': bookFilePath=%v, imageFilePath=%v, metadata=%v" $k (isset $entry "bookFilePath") (isset $entry "imageFilePath") (isset $entry "metadata") -}} {{- end -}} {{- end -}} {{- return $books -}}
